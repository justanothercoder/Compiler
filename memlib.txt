extern unsafe __brk(void* brk) : void*;
extern unsafe __mmap(int a, int size, int b, int c, int d, int e) : void*;

def unsafe sbrk(int size) : void*
{
	void* brk = __brk(null)
	
	__brk(brk + size);

	return brk;
}

def unsafe mmap(int size) : void* {
    return __mmap(0, size, 3, 34, 0-1, 0)
}
